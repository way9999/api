@app.route("/api/check-channels", methods=["POST"])
@require_auth
def api_check_channels():
    """Health check: test enabled Codex channels."""
    body = request.get_json(silent=True) or {}
    only_ids = body.get("ids") or []
    limit = int(body.get("limit") or 0)
    fast = body.get("fast", None)
    workers = body.get("workers", None)

    if isinstance(only_ids, (str, int)):
        only_ids = [only_ids]
    if isinstance(fast, str):
        fast = fast.lower() in ("1", "true", "yes")
    if isinstance(workers, str):
        try:
            workers = int(workers)
        except Exception:
            workers = None

    def _run_check():
        result = _check_channels_health(
            only_ids=only_ids,
            limit=limit,
            fast=fast,
            workers=workers,
        )
        broadcast_log(f"[health] check done: {json.dumps(result, ensure_ascii=False)}")
    threading.Thread(target=_run_check, daemon=True).start()
    return jsonify({"ok": True, "message": "health check started"})


def _check_channels_health():
    """Test status=1 type=57 channels against chatgpt.com."""
    from curl_cffi import requests as cffi_req

    summary = {"ok": 0, "deactivated": 0, "refreshed": 0, "disabled": 0,
               "usage_limit": 0, "reactivated": 0, "errors": 0}
    try:
        conn = _get_db()
        with conn.cursor() as cur:
            cur.execute("SELECT id, name, `key`, status FROM channels WHERE type = 57 AND status IN (1, 3)")
            rows = cur.fetchall() or []
        conn.close()
    except Exception as e:
        broadcast_log(f"[health] db error: {e}")
        return summary

    for row in rows:
        ch_id = row["id"]
        ch_name = row.get("name", f"#{ch_id}")
        token_data = _decode_channel_key(row.get("key"))
        access_token = token_data.get("access_token", "")
        account_id = token_data.get("account_id", "")
        email = token_data.get("email", "")

        if not access_token:
            continue

        try:
            headers = {
                "Authorization": f"Bearer {access_token}",
                "Content-Type": "application/json",
                "chatgpt-account-id": account_id,
                "OpenAI-Beta": "responses=experimental",
                "originator": "codex_cli_rs",
            }
            data = json.dumps({
                "model": "gpt-5-codex",
                "input": [{"role": "user", "content": "say hi"}],
                "instructions": "",
                "stream": True,
                "store": False,
            })
            resp = cffi_req.post(
                "https://chatgpt.com/backend-api/codex/responses",
                headers=headers, data=data, impersonate="chrome", timeout=30,
            )

            resp_text = resp.text[:500]

            # --- status=3 channels: check if recovered ---
            if int(row.get("status", 1)) == 3:
                if resp.status_code == 200 and "usage limit" not in resp_text.lower():
                    _update_channel_status(ch_id, 1)
                    broadcast_log(f"[health] {ch_name} е·ІжҒўеӨҚпјҢйҮҚж–°еҗҜз”Ё")
                    summary["reactivated"] += 1
                else:
                    broadcast_log(f"[health] {ch_name} д»ҚеҸ—йҷ?(status={resp.status_code})")
                continue

            # --- status=1 normal check ---
            if resp.status_code == 200:
                if "usage limit" in resp_text.lower():
                    _update_channel_status(ch_id, 3)
                    broadcast_log(f"[health] {ch_name} usage limitпјҢдёҙж—¶зҰҒз”?)
                    summary["usage_limit"] += 1
                else:
                    summary["ok"] += 1
                continue

            if resp.status_code == 401:
                if "deactivated" in resp_text.lower():
                    # account deactivated: delete channel + auth file
                    _delete_channel_and_auth(ch_id, email)
                    broadcast_log(f"[health] {ch_name} иҙҰеҸ·е·ІеҒңз”ЁпјҢе·ІеҲ йҷӨжё йҒ“е’Ң auth ж–Үд»¶")
                    summary["deactivated"] += 1
                elif "invalid" in resp_text.lower() or "expired" in resp_text.lower():
                    # token invalid: try refresh
                    if refresh_token:
                        try:
                            new_tokens = _refresh_access_token(refresh_token)
                            token_data["access_token"] = new_tokens["access_token"]
                            if new_tokens.get("refresh_token"):
                                token_data["refresh_token"] = new_tokens["refresh_token"]
                            if new_tokens.get("id_token"):
                                token_data["id_token"] = new_tokens["id_token"]
                            token_data["last_refresh"] = time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime())
                            token_data["expired"] = time.strftime(
                                "%Y-%m-%dT%H:%M:%SZ",
                                time.gmtime(time.time() + new_tokens.get("expires_in", 3600)),
                            )
                            new_key = json.dumps(token_data, ensure_ascii=False, separators=(",", ":"))
                            conn2 = _get_db()
                            with conn2.cursor() as cur2:
                                cur2.execute("UPDATE channels SET `key` = %s WHERE id = %s", (new_key, ch_id))
                                conn2.commit()
                            conn2.close()
                            broadcast_log(f"[health] {ch_name} token е·ІеҲ·ж–?)
                            summary["refreshed"] += 1
                        except Exception as re:
                            _update_channel_status(ch_id, 2)
                            broadcast_log(f"[health] {ch_name} еҲ·ж–°еӨұиҙҘ({re})пјҢе·ІзҰҒз”Ё")
                            summary["disabled"] += 1
                    else:
                        _update_channel_status(ch_id, 2)
                        broadcast_log(f"[health] {ch_name} ж—?refresh_tokenпјҢе·ІзҰҒз”Ё")
                        summary["disabled"] += 1
                else:
                    _update_channel_status(ch_id, 2)
                    broadcast_log(f"[health] {ch_name} 401 жңӘзҹҘеҺҹеӣ пјҢе·ІзҰҒз”Ё: {resp_text[:100]}")
                    summary["disabled"] += 1
                continue

            # other status codes
            if "usage limit" in resp_text.lower():
                _update_channel_status(ch_id, 3)
                broadcast_log(f"[health] {ch_name} usage limit (status={resp.status_code})пјҢдёҙж—¶зҰҒз”?)
                summary["usage_limit"] += 1
            else:
                broadcast_log(f"[health] {ch_name} ејӮеёё status={resp.status_code}: {resp_text[:100]}")
                summary["errors"] += 1

        except Exception as e:
            broadcast_log(f"[health] {ch_name} иҜ·жұӮејӮеёё: {e}")
            summary["errors"] += 1

    return summary
